                    {%
                    \def\FancyVerbFormatLine##1{\ifnum\value{FancyVerbLine}=1{\color{terminalCommandFg}##1}\else##1\fi}%

                        enhanced,
                        colback=codeBlockBg,

                        arc=3pt,
                        left=5pt,
                        right=5pt,

                        bottom=5pt,

                    \inputminted[fontsize=\footnotesize,breaklines=true,linenos=false]{text}{#1}%
                    \end{tcolorbox}%
                    }%
                \else
                    \begin{tcolorbox}[
                        enhanced,
                        colback=codeBlockBg,
                        colframe=codeBlockBorder,

                        right=5pt,

                        bottom=5pt,
