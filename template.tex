    rendererPrototypes={

        inlineHtmlTag = {

                    bottom=5pt,
                    breakable

                \end{tcolorbox}%
            \else
                \def\outputlang{output}%
                \ifx\templang\outputlang
                    {%
                    \def\FancyVerbFormatLine##1{\ifnum\value{FancyVerbLine}=1{\color{outputCommandFg}##1}\else##1\fi}%
                    \begin{tcolorbox}[
                        enhanced,
                        colback=codeBlockBg,
                        colframe=codeBlockBorder,
                        boxrule=0pt,
                        arc=3pt,
                        left=5pt,
                        right=5pt,
                        top=20pt,
                        bottom=5pt,
                        breakable,
                        overlay={\node[anchor=north east,font=\tiny\ttfamily,text=gray] at ([xshift=-5pt,yshift=-3pt]frame.north east) {#3};}
                    ]%
                    \inputminted[fontsize=\footnotesize,breaklines=true,linenos=false]{text}{#1}%
                    \end{tcolorbox}%
                    }%
                \else
                    \begin{tcolorbox}[
                        enhanced,
                        colback=codeBlockBg,
                        colframe=codeBlockBorder,
                        boxrule=0pt,
                        arc=3pt,
                        left=5pt,
                        right=5pt,
                        top=20pt,
                        bottom=5pt,
                        breakable,
                        overlay={\node[anchor=north east,font=\tiny\ttfamily,text=gray] at ([xshift=-5pt,yshift=-3pt]frame.north east) {#3};}

                    \end{tcolorbox}%
                \fi
            \fi
        },
